MISSION OVERVIEW
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Build a complete Python-based Browser Agent that replaces manual Selenium 
test scripting. The system has two phases:

LEARN phase: A human records themselves using a web app. The agent 
captures everything — page structure, form fields, interactions, 
navigation flows — and uses an LLM to convert this raw recording into 
structured test case definitions stored in a local SQLite database.

TEST phase: The agent autonomously executes those test cases using 
Playwright — filling forms, clicking buttons, asserting outcomes — 
and produces an HTML + JSON test report.

The key innovation: when the web app changes (e.g., a form goes from 
15 fields to 16 fields), the user simply re-records. The agent 
regenerates the test cases. No manual test script editing ever.

from pydantic import BaseModel
from typing import Optional
from datetime import datetime

class LocatorSpec(BaseModel):
    strategy: str        # "aria_label" | "role" | "placeholder" | "text" | 
                         # "id" | "css" | "xpath"
    value: str
    confidence: float    # 0.0 to 1.0

class ValidationRule(BaseModel):
    rule: str            # "required" | "email_format" | "min_length" | 
                         # "max_length" | "numeric"
    param: Optional[str] = None
    typical_error: Optional[str] = None

class ElementModel(BaseModel):
    id: str
    page_id: str
    element_type: str    # "input" | "button" | "select" | "link" | 
                         # "checkbox" | "radio" | "textarea"
    semantic_label: str  # LLM-generated: "Email address field"
    locators: list[LocatorSpec]
    validation_rules: list[ValidationRule]
    observed_values: list[str]

class PageModel(BaseModel):
    id: str
    version_id: str
    url_pattern: str     # regex: "^/login$" or "^/checkout"
    title: str
    purpose: str         # LLM-generated: "User authentication page"
    accessibility_snapshot: dict

class FlowStep(BaseModel):
    sequence: int
    action: str          # "fill" | "click" | "navigate" | "select" | 
                         # "check" | "hover" | "clear"
    element_id: Optional[str] = None
    value: Optional[str] = None
    url: Optional[str] = None

class ExpectedOutcome(BaseModel):
    type: str            # "navigation" | "element_visible" | "text_present" | 
                         # "element_absent"
    url_pattern: Optional[str] = None
    element_label: Optional[str] = None
    text: Optional[str] = None

class UserFlow(BaseModel):
    id: str
    version_id: str
    name: str            # LLM-generated: "User Login Flow"
    description: str
    start_url: str
    steps: list[FlowStep]
    expected_outcome: ExpectedOutcome

class TestStep(BaseModel):
    sequence: int
    action: str
    element_id: Optional[str] = None
    value: Optional[str] = None
    url: Optional[str] = None

class Assertion(BaseModel):
    type: str            # "url_contains" | "element_visible" | "text_equals" |
                         # "element_count" | "element_absent"
    expected: str
    element_label: Optional[str] = None

class TestCase(BaseModel):
    id: str
    flow_id: str
    name: str
    category: str        # "happy_path" | "negative" | "edge_case"
    steps: list[TestStep]
    assertions: list[Assertion]
    confidence: float
    generated_at: datetime

class StepResult(BaseModel):
    sequence: int
    action: str
    element_label: Optional[str] = None
    status: str          # "passed" | "failed" | "skipped"

    error: Optional[str] = None

Methods to implement:

async start_session():
  - Launch Playwright Chromium browser (headful, headless=False)
  - Create a new context and page
  - Navigate to start_url
  - Inject event_capture.js via page.evaluate() after each page load
  - Set up these Playwright event hooks on the page:
      page.on('framenavigated', self._on_navigate)
      page.on('dialog', self._on_dialog)
      page.on('request', self._on_request)
  - Start polling loop: every 500ms, call 
    page.evaluate("return window.__capturedEvents || []") 
    and drain into self._event_buffer
    then call page.evaluate("window.__capturedEvents = []") to clear

async _on_navigate(frame):
  - Only process main frame navigations
  - Capture: current URL, page title
  - Capture accessibility snapshot: await page.accessibility.snapshot()
  - Take screenshot (use utils/screenshots.py)
  - Store all of above as a NavigationEvent in self._navigation_log

async _on_dialog(dialog):
  - Log the dialog type and message
  - Auto-dismiss: await dialog.dismiss()

async _on_request(request):
  - Log: method, url (only if it contains /api/ to reduce noise)

async stop_session():
  - Stop the polling loop
  - Close browser
  - Return a dict: {
      "events": self._event_buffer,
      "navigations": self._navigation_log,
      "app_name": self.app_name,
      "start_url": self.start_url,
      "recorded_at": ISO timestamp
    }

async save_session(output_path: str):
  - Call stop_session()
  - Write returned dict to output_path as formatted JSON

Implement class LLMClient exactly as follows:

```python
import os
import json
import hashlib
from groq import AsyncGroq
import groq as groq_module
from tenacity import (
    retry, stop_after_attempt,
    wait_exponential, retry_if_exception_type
)
from llm.cache import LLMCache

class LLMClient:
    def __init__(self):
        self.client = AsyncGroq(api_key=os.environ.get("GROQ_API_KEY"))
        self.smart_model = os.environ.get(
            "GROQ_SMART_MODEL", "llama-3.3-70b-versatile"
        )
        self.fast_model = os.environ.get(
            "GROQ_FAST_MODEL", "llama-3.1-8b-instant"
        )
        self.cache = LLMCache()

    async def generate(
        self,
        prompt: str,
        model: str = "haiku",   # accepts "haiku" (fast) or "sonnet" (smart)
        expect_json: bool = True
    ) -> dict | str:

        # Map familiar names to actual Groq model strings
        actual_model = (
            self.fast_model if model == "haiku" else self.smart_model
        )

        # Check cache before calling API
        input_hash = hashlib.sha256(
            f"{actual_model}{prompt}".encode()
        ).hexdigest()
        cached = await self.cache.get(input_hash)
        if cached:
            return json.loads(cached) if expect_json else cached

        # Call Groq API
        response = await self._call_api(actual_model, prompt, expect_json)
        text = response.choices[0].message.content.strip()

        if expect_json:
            text = self._strip_json_fences(text)
            try:
                result = json.loads(text)
            except json.JSONDecodeError:
                # Second attempt: ask the model to fix its own output
                fix_prompt = (
                    f"The following is not valid JSON. "
                    f"Fix it and return ONLY valid JSON:\n{text}"
                )
                retry_response = await self._call_api(
                    actual_model, fix_prompt, expect_json=True
                )
                text = self._strip_json_fences(
                    retry_response.choices[0].message.content.strip()
                )
                result = json.loads(text)

            await self.cache.set(input_hash, json.dumps(result), actual_model)
            return result
        else:
            await self.cache.set(input_hash, text, actual_model)
            return text

    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=2, max=10),
        retry=retry_if_exception_type((
            groq_module.RateLimitError,
            groq_module.APIStatusError,
            groq_module.APIConnectionError
        ))
    )
    async def _call_api(self, model: str, prompt: str, expect_json: bool):
        kwargs = dict(
            model=model,
            messages=[
                {
                    "role": "system",
                    "content": (
                        "You are a precise assistant. "
                        "When asked to return JSON, respond ONLY with "
                        "valid JSON — no preamble, no explanation, "
                        "no markdown code fences."
                    )
                },
                {"role": "user", "content": prompt}
            ],
            max_tokens=2000,
            temperature=0.1   # low temperature = more consistent JSON output
        )
        if expect_json:
            kwargs["response_format"] = {"type": "json_object"}
        return await self.client.chat.completions.create(**kwargs)

    @staticmethod
    def _strip_json_fences(text: str) -> str:
        """Remove markdown code fences if the model includes them anyway."""
        if text.startswith("```"):
            lines = text.split("\n")
            lines = [l for l in lines if not l.startswith("```")]
            text = "\n".join(lines)
        return text.strip()
```

IMPORTANT NOTES FOR THE AGENT:
- response_format={"type": "json_object"} is supported by both 
  llama-3.3-70b-versatile and llama-3.1-8b-instant on Groq.
  Always pass it when expect_json=True — this is more reliable than 
  prompting alone.
- temperature=0.1 is intentional. Do not increase it. Higher 
  temperatures cause inconsistent JSON structure.
- The model parameter in generate() accepts "haiku" or "sonnet" 
  to match the rest of the codebase. Do not change the interface.
- Groq's free tier has rate limits (roughly 30 RPM on the 70B model,
  6000 RPM on the 8B model). The tenacity retry handles RateLimitError
  automatically with exponential backoff.

Implement two functions:

def generate_locators(element_attrs: dict) -> list[LocatorSpec]:
  Given raw element attributes from the JS event capture, generate
  a ranked list of LocatorSpec objects in this priority order:
  1. aria_label  (confidence 0.95) — only if element_attrs["aria_label"]
  2. placeholder (confidence 0.85) — only if element_attrs["placeholder"]
  3. role+text   (confidence 0.80) — if button or link with text_content
  4. id          (confidence 0.70) — only if id is not empty and not 
                                     auto-generated (no UUIDs, no numbers only)
  5. css_name    (confidence 0.55) — input[name="X"] or button[type="submit"]
  6. xpath_text  (confidence 0.40) — //button[contains(text(),"X")]

  An ID is "auto-generated" if it matches: r'^[0-9]+$' or 
  r'^[a-f0-9]{8}-' or r'^react-' or r'^ember'

async def resolve_locator(page, element: ElementModel, llm_client,
                          timeout_ms: int = 3000) -> tuple[any, str]:
  Try each locator in element.locators sorted by confidence descending.
  For each locator:
    - Build the Playwright locator using the strategy
    - Try: await expect(loc).to_be_visible(timeout=timeout_ms)
    - If visible: return (locator, locator_spec.strategy)
    - If PlaywrightError: continue to next

  If all fail:
    - Get current accessibility tree: await page.accessibility.snapshot()
    - Call llm_client with LOCATOR_FALLBACK_PROMPT
    - Build locator from LLM response
    - Return (locator, "llm_fallback")

  If LLM fallback also fails: raise ElementNotFoundError

For building Playwright locators from LocatorSpec:
  "aria_label"  → page.get_by_label(value)
  "placeholder" → page.get_by_placeholder(value)
  "role"        → page.get_by_role(role, name=name)
  "text"        → page.get_by_text(value, exact=False)
  "id"          → page.locator(f"#{value}")
  "css_name"    → page.locator(value)
  "xpath_text"  → page.locator(value)

async run_test(test_case: TestCase, elements: dict[str, ElementModel],
               run_id: str) -> TestResult:
  - Launch browser context (reuse across tests with context pool)
  - Navigate to flow start_url
  - For each step:
      a. If action is "navigate": await page.goto(url)
      b. Otherwise: resolve element locator, execute action
      c. After action: wait for networkidle (timeout 10s)
      d. Take screenshot via utils/screenshots.py
      e. Append StepResult
  - Call verifier.evaluate_assertions(page, test_case.assertions)
  - Build and return TestResult

Action execution rules:
  "fill"     → await locator.fill(value)
  "click"    → await locator.click()
  "check"    → await locator.check()
  "select"   → await locator.select_option(value)
  "hover"    → await locator.hover()
  "clear"    → await locator.clear()
  "navigate" → await page.goto(step.url)

Error handling:
  - If ElementNotFoundError: StepResult status="failed", continue to next test
  - If any other PlaywrightError: StepResult status="errored", continue
  - Never raise from run_test — always return a TestResult
